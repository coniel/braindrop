## Usage

Import the component from the `@minddrop/ui` package.

```jsx
import { NavGroup, TopicNavItem } from '@minddrop/ui';

export default () => {
  return (
    <NavGroup title="Topics">
      <TopicNavItem />
      <TopicNavItem />
      <TopicNavItem />
    </NavGroup>
  );
};
```
