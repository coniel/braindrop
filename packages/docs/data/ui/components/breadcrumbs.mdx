## Usage

Import the component from the `@minddrop/ui` package.

```jsx
import { Breadcrumbs, Breadcrumb } from '@minddrop/ui';

export default () => {
  return (
    <Breadcrumbs>
      <Breadcrumb label="Tea" />
      <Breadcrumb label="Books" />
      <Breadcrumb label="The Book Of Tea" />
    </Breadcrumbs>
  );
};
```
