## Usage

Import the component from the `@minddrop/ui` package.

```jsx
import { Sidebar } from '@minddrop/ui';

export default ({ handleSidebarResized }) => {
  return (
    <Sidebar
      initialWidth={275}
      minWidth={200}
      maxWidth={400}
      onResized={handleSidebarResized}
    >
      ...
    </Sidebar>
  );
};
```
