## Usage

### Using ContextMenuContent's content prop

Context menus can be generated by passing an array of item descriptors as the `ContextMenuContent`'s `content` prop.

```jsx
import {
  IconButton,
  ContextMenu,
  ContextMenuTrigger,
  ContextMenuContent,
  MenuContent,
} from '@minddrop/ui';
import {
  MoreVertical,
  Title,
  Note,
  TurnInto,
  Link,
  Move,
  AddTo,
} from '@minddrop/icons';

const menu: MenuContent = [
  {
    label: 'Add title',
    icon: <Title />,
    onSelect: () => console.log('Add title'),
    keyboardShortcut: ['Ctrl', 'T'],
  },
  {
    label: 'Add note',
    icon: <Note />,
    onSelect: () => console.log('Add note'),
    keyboardShortcut: ['Ctrl', 'Shift', 'N'],
  },
  {
    label: 'Turn into',
    icon: <TurnInto />,
    submenu: [
      {
        label: 'Text',
        onSelect: () => console.log('Turn into text'),
      },
      {
        label: 'Image',
        onSelect: () => console.log('Turn into image'),
      },
      {
        label: 'Equation',
        onSelect: () => console.log('Turn into equation'),
      },
    ],
  },
  '---',
  'Actions',
  {
    label: 'Copy link',
    icon: <Link />,
    onSelect: () => console.log('Copy link'),
    keyboardShortcut: ['Ctrl', 'Shift', 'C'],
  },
  {
    label: 'Move to',
    icon: <MoveTo />,
    onSelect: () => console.log('Move to'),
    keyboardShortcut: ['Ctrl', 'M'],
    tooltipTitle: 'Move drop to another topic',
  },
  {
    label: 'Add to',
    icon: <AddTo />,
    onSelect: () => console.log('Add to'),
    keyboardShortcut: ['Ctrl', 'A'],
    tooltipTitle: 'Add drop to another topic',
    tooltipDescription: 'Drops appearing in multiple topics are kept in sync.',
  },
];

export default () => {
  return (
    <ContextMenu>
      <ContextMenuTrigger>
        <IconButton label="Drop options">
          <MoreVertical>
        </IconButton>
      </ContextMenuTrigger>
      <ContextMenuContent content={menu} />
    </ContextMenu>
  );
};
```

### Composing a menu

You can also compose a menu yourself using the available components. The menu below is equivalent to the one generated by the example above.

```jsx
import {
  ContextMenu,
  ContextMenuTrigger,
  ContextMenuContent,
  ContextMenuLabel,
  ContextMenuSeparator,
  ContextMenuItem,
  ContextMenuTriggerItem,
} from '@minddrop/ui';
import {
  Title,
  Note,
  TurnInto,
  Link,
  Move,
  AddTo,
} from '@minddrop/icons';

export default () => {
  return (
    <ContextMenu>
      <ContextMenuTrigger>
        <Drop />
      </ContextMenuTrigger>
      <ContextMenuContent>
        <ContextMenuItem
          label="Add title"
          icon={<Title />}
          keyboardShortcut={['Ctrl', 'T']}
          onSelect={handleAddTitle}
        />
        <ContextMenuItem
          label="Add note"
          icon={<Note />}
          keyboardShortcut={['Ctrl', 'Shift', 'N']}
          onSelect={handleAddNote}
        />
        <ContextMenu>
          <ContextMenuTriggerItem
            label="Turn into"
            icon={<TurnInto />}
          />
          <ContextMenuContent>
            <ContextMenuItem label="Text" onSelect={turnIntoText} />
            <ContextMenuItem label="Image" onSelect={turnIntoImage} />
            <ContextMenuItem
              label="Equation"
              onSelect={turnIntoEquation}
            />
          </ContextMenuContent>
        </ContextMenu>
        <ContextMenuSeparator />
        <ContextMenuLabel>Actions</ContextMenuLabel>
        <ContextMenuItem
          label="Copy link"
          icon={<Copy />}
          keyboardShortcut={['Ctrl', 'Shift', 'C']}
          onSelect={handleCopyLink}
        />
        <ContextMenuItem
          label="Move to"
          icon={<MoveTo />}
          keyboardShortcut={['Ctrl', 'M']}
          onSelect={handleMoveTo}
          tooltipTitle="Move drop to another topic"
        />
        <ContextMenuItem
          label="Add to"
          icon={<AddTo />}
          keyboardShortcut={['Ctrl', 'A']}
          onSelect={handleAddTo}
          tooltipTitle="Add drop to another topic"
          tooltipDescription="Drops appearing in multiple topics are kept in sync."
        />
      </ContextMenuContent>
    </ContextMenu>
  );
};
```

## API Reference

Context menus can be generated by passing an array as the `ContextMenuContent`'s `content` prop.

### Label

To create a `ContextMenuLabel`, simply add the label as a string to the menu array.

```jsx
const menu = [
  ...
  'Actions'
  ...
]
```

### Separator

To create a `ContextMenuSeparator`, add three minus signs `---` to the menu array.

```jsx
const menu = [
  ...
  '---'
  ...
]
```

## Styling

### Origin-aware animations

Radix exposes a CSS custom property `--radix-context-menu-content-transform-origin`. Use it to animate the content from its computed origin based on `side`, `sideOffset`, `align`, `alignOffset` and any collisions.

```css line=14-16
@keyframes scaleIn {
  from {
    opacity: 0;
    transform: scale(0);
  }

  to {
    opacity: 1;
    transform: scale(1);
  }
}

.context-menu {
  transform-origin: var(
    --radix-context-menu-content-transform-origin
  );
  animation: scaleIn 0.5s ease-out;
}
```

### Collision-aware animations

Radix exposes `data-side` and `data-align` attributes. Their values will change at runtime to reflect collisions. Use them to create collision and direction-aware animations.

```css line=30-32,34-36
@keyframes slideDown {
  from {
    opacity: 0;
    transform: translateY(-10px);
  }

  to {
    opacity: 1;
    transform: translateY(0);
  }
}

@keyframes slideUp {
  from {
    opacity: 0;
    transform: translateY(0);
  }

  to {
    opacity: 1;
    transform: translateY(-10px);
  }
}

.context-menu {
  animation-duration: 0.6s;
  animation-timing-function: cubic-bezier(0.16, 1, 0.3, 1);
}

.context-menu[data-side='bottom'] {
  animation: slideDown;
}

.context-menu[data-side='top'] {
  animation: slideUp;
}
```

## Accessibility

Adheres to the [Menu WAI-ARIA design pattern](https://www.w3.org/TR/wai-aria-practices-1.2/#menu) and uses [roving tabindex](https://www.w3.org/TR/wai-aria-practices-1.2/examples/menu-button/menu-button-actions.html) to manage focus movement among menu items.
