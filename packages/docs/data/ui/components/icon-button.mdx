## Usage

Import the component from the `@minddrop/ui` package.

```jsx
import { IconButton } from '@minddrop/ui';
import { Close } from '@minddrop/icons';

export default () => {
  return (
    <IconButton label="Close">
      <Close />
    </IconButton>
  );
};
```
