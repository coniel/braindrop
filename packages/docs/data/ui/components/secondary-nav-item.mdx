## Usage

Import the component from the `@minddrop/ui` package.

```jsx
import { SecondaryNavItem, Icon } from '@minddrop/ui';

export default () => {
  return (
    <SecondaryNavItem label="Trash" icon={<Icon name="trash" />} />
  );
};
```
